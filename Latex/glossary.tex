\newglossaryentry{nRF52}
{
	name=nRF52,
    description={Nordic Semiconductor nRF52 DK , Development Kit for the nRF52 microcontroller used as end nodes in the testbed. From now on noted as "nRF52"}
}

\newglossaryentry{Raspberry Pi}
{
	name=Raspberry Pi,
    description={Mini computer in the size of a credit card. Runs on electric power from a cord in the testbed. From now on noted as "Raspberry Pi" or shortened to "Pi"}
}

\newglossaryentry{ADXL345}
{
	name=ADXL345 accelerometer,
    description={Small accelerometer mounted on a chip delivered by Adafruit. Connected to the nRF52 in the testbed}
}


\newglossaryentry{payload}
{
    name=payload,
    description={The part of the transmitted data that is the intended message}
}

\newglossaryentry{goodput}
{
	name=goodput,
	description={The number of bytes in a packet that contains the intended message, sent one way through the network per time unit}
}

\newglossaryentry{throughput}
{
	name=throughput,
	description={The total amount of data sent one way through the network per time unit. The sum of payload and additional header files needed to transport the packet}
}
